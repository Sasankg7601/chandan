Shift-Reduce Parser in C

#include <stdio.h>
#include <string.h>

char stack[50], input[50];
int top = -1, i = 0;

void check() {
    if (top >= 2 && stack[top]=='T' && stack[top-1]=='+' && stack[top-2]=='E') {
        top -= 2; stack[top] = 'E';
        printf("Reduce: E→E+T\n");
    }
    else if (stack[top]=='T') {
        stack[top] = 'E';
        printf("Reduce: E→T\n");
    }
    else if (top >= 2 && stack[top]=='F' && stack[top-1]=='*' && stack[top-2]=='T') {
        top -= 2; stack[top] = 'T';
        printf("Reduce: T→T*F\n");
    }
    else if (stack[top]=='F') {
        stack[top] = 'T';
        printf("Reduce: T→F\n");
    }
    else if (top >= 2 && stack[top]==')' && stack[top-1]=='E' && stack[top-2]=='(') {
        top -= 2; stack[top] = 'F';
        printf("Reduce: F→(E)\n");
    }
    else if (top >= 1 && stack[top]=='d' && stack[top-1]=='i') {
        top--; stack[top] = 'F';
        printf("Reduce: F→id\n");
    }
}

int main() {
    printf("Enter input string (use id for identifiers): ");
    scanf("%s", input);

    printf("\nShift Reduce Parsing Steps:\n----------------------------\n");
    while (1) {
        if (i < strlen(input)) {
            stack[++top] = input[i++];
            printf("Shift: %c\n", stack[top]);
            check();
        } else {
            check();
            if (top == 0 && stack[top] == 'E') {
                printf("ACCEPT\n");
                break;
            } else if (i >= strlen(input)) {
                printf("REJECT\n");
                break;
            }
        }
    }
    return 0;
}
